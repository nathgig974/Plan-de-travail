\newcommand{\bulle}[3]{
   \psframe[linewidth=1mm,framearc=0.1,linecolor=#1](0,0)(5,3)
   \psline[linecolor=#1](0,2.2)(5,2.2)
   \rput(2.5,2.6){#2}
   \rput(2.5,1.1){\begin{minipage}{4cm}
                           {\small\darkgray #3}
                         \end{minipage}}}
